in.js

in.js is a javascript loader, by using it, you can load your js/css file asynchronous and non-blocking at any time.

v0.1.9

IN.JS IN THE INDUSTRY

EXAMPLES:

` Add Modules 
In.add('mod1',{path:'mod1.js',type:'js',charset:'utf-8'});
In.add('mod2',{path:'mod2.js',type:'js',charset:'utf-8',rely:['mod3']});
In.add('mod3',{path:'mod3.js',type:'js',charset:'utf-8'});
` Load Modules 
In('mod1','mod2',function() {
	//inline function
},function() {
	//callback
});
` Load Modules After DOM Ready 
In.ready('mod1','mod2',function() {
	console.log($);
});
` Load Modules after 3000ms 
In.later(3000,'mod1','mod2',function() {
	console.log($);
});
` Insert CSS Style Dynamic 
In.css('body {background:yellow}');
